\subsection*{2.}
$(p \land \neg q) \rightarrow \neg r \equiv (p \land r) \rightarrow q$

$$
\begin{array}{|c|c|}
\hline
(p \land \neg q) \rightarrow \neg r & \\
\hline
\neg (p \land \neg q) \lor \neg r & \text{Conditional identity} \\
\hline
\neg p \lor \neg \neg q \lor \neg r & \text{DeMorgan's law} \\
\hline
\neg p \lor q \lor \neg r & \text{Double negation law} \\
\hline
\neg p \lor \neg r \lor q & \text{Commutative law} \\
\hline
\neg (p \land r) \lor q & \text{DeMorgan's law} \\
\hline
(p \land r) \rightarrow q & \text{Conditional identity} \\
\hline
\end{array}
$$
