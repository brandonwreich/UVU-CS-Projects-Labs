\subsection*{1.9.4}
\begin{itemize}
    \item c - $\forall x \exists y(P(x,y) \land \neg Q(x,y))$
      $$
        \begin{array}{|c|c|}
        \hline
        \exists x \forall y(P(x,y) \rightarrow Q(x,y)) & \\
        \hline
        \neg \exists x \forall y(P(x,y) \rightarrow Q(x,y)) & \text{Negation} \\
        \hline
        \neg \exists x \forall y(\neg P(x,y) \lor Q(x,y)) & \text{Biconditional Identity} \\
        \hline
        \forall x \exists y \neg (\neg P(x,y) \lor Q(x,y)) & \text{De Morgan's Law} \\
        \hline
        \forall x \exists y (\neg \neg P(x,y) \land \neg Q(x,y)) & \text{De Morgan's Law} \\
        \hline
        \forall x \exists y (P(x,y) \land \neg Q(x,y)) & \text{Double Negation Law} \\
        \hline
        \end{array}
      $$
    \item e - $\forall x \forall y \neg P(x,y) \lor \exists x \exists y \neg Q(x,y)$
     $$
       \begin{array}{|c|c|}
       \hline
       \exists x \exists y P(x,y) \land \forall x \forall y Q(x,y) & \\
       \hline
       \neg (\exists x \exists y P(x,y) \land \forall x \forall y Q(x,y)) & \text{Negation} \\
       \hline
       \neg \exists x \exists y P(x,y) \lor \neg \forall x \forall y Q(x,y) & \text{De Morgan's Law} \\
       \hline
       \forall x \forall y \neg P(x,y) \lor \exists x \exists y \neg Q(x,y) & \text{De Morgan's Law} \\
       \hline
       \end{array}
     $$
